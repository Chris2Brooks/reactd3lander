import { graphql } from "gatsby"
import { ArticleListing } from "../../components/ArticleListing"
import { Container } from '../../components/Blocks/index'

export const pageQuery = graphql`
  query {
    allSitePage(
      filter: { path: { regex: "/articles/.+/" } }
      sort: { fields: context___frontmatter___date, order: DESC }
    ) {
      nodes {
        path
        context {
          frontmatter {
            title
            description
            date
          }
        }
      }
    }
  }
`

<Container width={[1, 3/4, 1/2]}>

Also signup for the mailing list etc.

<>
  {props.data.allSitePage.nodes.map((props, i) => (
    <ArticleListing {...props} key={i} />
  ))}
</>

</Container>
